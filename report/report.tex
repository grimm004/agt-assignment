			% second set
			\node at (2,\xitem) (b\xitem) {};   
		}

		\foreach \x [count=\xi] in {#2}
		{
			\foreach \tritem in \x
			\draw(a\xi) -- (b\tritem);
		}
    \end{tikzpicture}
